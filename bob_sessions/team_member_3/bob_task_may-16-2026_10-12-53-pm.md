 53 | def get_session(session_id: str) -> Optional[Dict]:
 54 |     """
 55 |     Get session by ID
 56 |     
 57 |     Args:
 58 |         session_id: Session ID
 59 |     
 60 |     Returns:
 61 |         Session dictionary or None if not found/expired
 62 |     """
 63 |     if session_id not in _sessions:
 64 |         return None
 65 |     
 66 |     session = _sessions[session_id]
 67 |     
 68 |     # Check if session has expired
 69 |     last_accessed = session.get("lastAccessedAt")
 70 |     if last_accessed:
 71 |         expiry_time = last_accessed + timedelta(minutes=SESSION_EXPIRY_MINUTES)
 72 |         if datetime.utcnow() > expiry_time:
 73 |             # Session expired, remove it
 74 |             del _sessions[session_id]
 75 |             return None
 76 |

 78 |     session["lastAccessedAt"] = datetime.utcnow()
 79 |     
 80 |     return session

 84 |     """
 85 |     Update session with new data
 86 |     
 87 |     Args:
 88 |         session_id: Session ID

 91 |     Returns:
